GET /api/v1/action-items — List with Filters

Auth: Required + injectTenant

Query Params:

?assigneeId=usr_01
?completed=false
?priority=HIGH&priority=URGENT   (multi-value)
?meetingId=mtg_01
?hasJiraTicket=true
?from=2026-05-01
?search=payment bug
?limit=20&cursor=cursor_abc
?sortBy=priority&sortOrder=desc

Service Logic: Similar to commitments list — cursor paginated with dynamic WHERE clause.

Response per item:

  "text":     "Fix payment bug in checkout flow",

  "completed": false,
  "assignee": {
    "id":   "usr_02",

PATCH /api/v1/action-items/:actionItemId — Update

Auth: Required + injectTenant

Request Body (all optional):

json{
  "completed":  true,
  "assigneeId": "usr_03",
  "dueDate":    "2026-05-20T23:59:59Z",
  "priority":   "URGENT",
  "text":       "Fix payment bug (updated)"
}

Service Logic:

1. Verify actionItem.teamId === req.teamId
2. If completed=true: set completedAt + completedById
3. If assigneeId changed: verify new assignee is team member
4. Update in DB
5. If Jira linked + completed: queue integrate job to mark Jira Done
6. Return updated item


POST /api/v1/action-items/:actionItemId/sync — Sync to Jira/Linear/Notion

Auth: Required + injectTenant

Request Body:

json{ "provider": "JIRA" }

Service Logic:

1. Check integration connected for provider → 422 if not
2. Queue integrate job immediately (high priority)
3. Return: { provider, status: "queued", message: "Sync started" }

Full Jira sync implementation comes Day 58. Today: queue the job, return immediately.


GET /api/v1/action-items/my — Current User's Action Items

Auth: Required + injectTenant

Filtered to assigneeId = req.user.id. Sorted by priority DESC, dueDate ASC.


notify.worker.ts — Notification Routing (Filled In)

File: services/api/src/queues/workers/notify.worker.ts

typescriptexport const notifyWorker = new Worker<NotifyJobData>(
  'notify',
  async (job: Job<NotifyJobData>) => {
    const { type, teamId, commitmentId, ownerId, meetingId, managerIds } = job.data

    // Load user notification preferences
    const ownerPrefs = ownerId
      ? await prisma.notificationPreference.findUnique({ where: { userId: ownerId } })
      : null

    // Load Slack integration for team
    const slackIntegration = await prisma.teamIntegration.findFirst({
      where: { teamId, provider: 'SLACK', isActive: true },
    })

    // Dedup check
    const dedupKey = `notif:dedup:${type}:${ownerId ?? teamId}:${commitmentId ?? meetingId ?? 'global'}`
    if (await redis.exists(dedupKey)) return
    await redis.setex(dedupKey, 3600, '1')

    switch (type) {
      case 'MEETING_PROCESSED': {
        const meeting = meetingId ? await prisma.meeting.findUnique({ where: { id: meetingId } }) : null
        if (!meeting) return

        // Send Slack summary to team channel
        if (slackIntegration) {
          // Full Slack implementation: Day 60
          logger.info({ meetingId }, 'notify: Slack meeting summary queued (Day 60)')
        }

        // Send email summary to managers
        if (managerIds?.length) {
          for (const managerId of managerIds) {
            const manager = await prisma.user.findUnique({ where: { id: managerId } })
            if (!manager) continue
            await emailService.sendMeetingSummary({ to: manager.email, meeting })
          }
        }
        break
      }

      case 'COMMITMENT_MISSED': {
        if (!commitmentId || !ownerId) return
        const [commitment, owner] = await Promise.all([
          prisma.commitment.findUnique({ where: { id: commitmentId } }),
          prisma.user.findUnique({ where: { id: ownerId } }),
        ])
        if (!commitment || !owner) return

        // Email to owner
        if (ownerPrefs?.preferences?.email?.commitmentMissed !== false) {
          await emailService.sendCommitmentMissed({ to: owner.email, commitment })
        }

        // Alert managers
        if (managerIds?.length) {
          for (const managerId of managerIds) {
            const manager = await prisma.user.findUnique({ where: { id: managerId } })
            if (manager) {
              await emailService.sendManagerAlert({ to: manager.email, owner, commitment })
            }
          }
        }
        break
      }

      case 'DEADLINE_REMINDER': {
        if (!commitmentId || !ownerId) return
        const [commitment, owner] = await Promise.all([
          prisma.commitment.findUnique({ where: { id: commitmentId } }),
          prisma.user.findUnique({ where: { id: ownerId } }),
        ])
        if (!commitment || !owner) return

        if (ownerPrefs?.preferences?.email?.deadlineReminder !== false) {
          await emailService.sendDeadlineReminder({ to: owner.email, commitment })
        }
        break
      }
    }
  },
  {
    connection: { host: process.env.REDIS_HOST, port: parseInt(process.env.REDIS_PORT ?? '6379') },
    concurrency: 5,
  }
)

useOnboarding.ts Hook

File: apps/web/src/features/onboarding/hooks/useOnboarding.ts

typescriptexport function useOnboarding() {
  const router = useRouter()
  const { user, setUser } = useAuthStore()

  const createTeam = useMutation({
    mutationFn: (data: { name: string; slug?: string }) =>
      apiClient.post('/api/v1/teams', data).then((r) => r.data.data),
    onSuccess: (team) => {
      // Update local user state with new teamId
      setUser({ ...user!, teamId: team.id })
      router.push('/onboarding/invite-team')
    },
  })

  const inviteMembers = useMutation({
    mutationFn: (emails: string[]) =>
      apiClient.post('/api/v1/teams/me/invite', { emails, role: 'MEMBER' })
        .then((r) => r.data.data),
    onSuccess: () => router.push('/onboarding/connect-calendar'),
  })

  const skipInvite = () => router.push('/onboarding/connect-calendar')

  const completeOnboarding = useMutation({
    mutationFn: () =>
      apiClient.patch('/api/v1/auth/me', { onboardingCompleted: true }).then((r) => r.data.data),
    onSuccess: () => router.push('/dashboard'),
  })

  return { createTeam, inviteMembers, skipInvite, completeOnboarding }
}

router.get('/google-calendar', requireAuth, asyncHandler(async (req, res) => {
  const state = crypto.randomBytes(32).toString('hex')
  await redis.setex(`oauth:state:calendar:${state}`, 600, req.user.id)

  const params = new URLSearchParams({
    client_id:     process.env.GOOGLE_CLIENT_ID!,
    redirect_uri:  `${process.env.API_URL}/auth/google-calendar/callback`,
    response_type: 'code',
    scope:         'https://www.googleapis.com/auth/calendar.readonly',
    state,
    access_type:   'offline',
    prompt:        'consent',
  })
  res.redirect(`https://accounts.google.com/o/oauth2/v2/auth?${params}`)
}))
