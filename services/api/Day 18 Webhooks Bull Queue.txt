Recall.ai Webhook Handler

File: services/api/src/modules/webhooks/recall.webhook.ts

typescriptimport { createHmac, timingSafeEqual } from 'crypto'
import { Request, Response } from 'express'
import { logger } from '../../config/logger'
import { updateMeetingStatus } from '../meetings/meetings.service'
import { transcribeQueue } from '../../queues/queue.client'
import { mongoService } from '../../services/mongo.service'
import { redis } from '../../config/redis'

// ── SIGNATURE VERIFICATION ───────────────────────────────────────────────────

function verifyRecallSignature(req: Request): void {
  const signature = req.headers['x-recall-signature'] as string
  if (!signature) {
    throw new Error('Missing X-Recall-Signature header')
  }

  // rawBody must be preserved — configure Express before json() middleware for this route
  const rawBody   = (req as any).rawBody as Buffer
  const expected  = createHmac('sha256', process.env.RECALL_WEBHOOK_SECRET!)
    .update(rawBody)
    .digest('hex')

  const sigBuf = Buffer.from(signature.replace('sha256=', ''))
  const expBuf = Buffer.from(expected)

  if (sigBuf.length !== expBuf.length || !timingSafeEqual(sigBuf, expBuf)) {
    throw new Error('Invalid Recall.ai webhook signature')
  }
}

// ── WEBHOOK HANDLER ──────────────────────────────────────────────────────────

export async function handleRecallWebhook(req: Request, res: Response): Promise<void> {
  // 1. Verify signature FIRST — reject immediately if invalid
  try {
    verifyRecallSignature(req)
  } catch {
    logger.warn('Recall.ai webhook signature verification failed')
    res.status(400).json({ received: false, error: 'Invalid signature' })
    return
  }

  // 2. Acknowledge immediately — Recall.ai expects < 5s response
  res.status(200).json({ received: true })

  // 3. Process asynchronously (never block the HTTP response)
  const { event, data } = req.body

  // Idempotency check — Recall.ai may send duplicate events
  const idempotencyKey = `webhook:processed:recall:${data.bot_id}:${event}`
  const alreadyProcessed = await redis.exists(idempotencyKey)
  if (alreadyProcessed) {
    logger.info({ event, botId: data.bot_id }, 'Recall webhook already processed — skipping')
    return
  }
  await redis.setex(idempotencyKey, 86400, '1')

  // 4. Route event to handler
  try {
    switch (event) {
      case 'bot.joining_call':
        await handleBotJoining(data.bot_id)
        break

      case 'bot.in_waiting_room':
        // No status change — just log. Bot waiting for host to admit.
        logger.info({ botId: data.bot_id }, 'Bot in waiting room')
        break

      case 'bot.recording_started':
        await handleRecordingStarted(data.bot_id)
        break

      case 'bot.done':
        await handleBotDone(data)
        break

      case 'bot.failed':
        await handleBotFailed(data.bot_id, data.reason)
        break

      default:
        logger.warn({ event }, 'Unknown Recall.ai event — ignoring')
    }
  } catch (err) {
    logger.error({ event, err }, 'Error processing Recall.ai webhook')
    // Don't throw — response already sent. Log for investigation.
  }
}

// ── EVENT HANDLERS ────────────────────────────────────────────────────────────

async function handleBotJoining(botId: string): Promise<void> {
  const meeting = await findMeetingByBotId(botId)
  if (!meeting) return
  await updateMeetingStatus(meeting.id, 'BOT_JOINING')

  // Real-time: emit Socket.io event to team dashboard
  const { getIO } = await import('../../realtime/socket.server')
  getIO()?.to(`team:${meeting.teamId}`).emit('meeting:bot_joining', {
    meetingId: meeting.id,
  })
}

async function handleRecordingStarted(botId: string): Promise<void> {
  const meeting = await findMeetingByBotId(botId)
  if (!meeting) return
  await updateMeetingStatus(meeting.id, 'RECORDING', { startedAt: new Date() })

  const { getIO } = await import('../../realtime/socket.server')
  getIO()?.to(`team:${meeting.teamId}`).emit('meeting:recording', {
    meetingId: meeting.id,
    startedAt: new Date().toISOString(),
  })
}

async function handleBotDone(data: RecallBotDonePayload): Promise<void> {
  const meeting = await findMeetingByBotId(data.bot_id)
  if (!meeting) {
    logger.error({ botId: data.bot_id }, 'bot.done received for unknown bot')
    return
  }

  const now             = new Date()
  const durationMinutes = meeting.startedAt
    ? Math.round((now.getTime() - meeting.startedAt.getTime()) / 60000)
    : 0

  // 1. Update meeting: PROCESSING + timing
  await updateMeetingStatus(meeting.id, 'PROCESSING', {
    endedAt:         now,
    durationMinutes,
  })

  // 2. Store raw transcript in MongoDB
  const mongoTranscriptId = await mongoService.storeTranscript({
    meeting_id:        meeting.id,
    team_id:           meeting.teamId,
    recall_bot_id:     data.bot_id,
    platform:          meeting.platform.toLowerCase(),
    raw_transcript:    data.transcript ?? [],
    full_text:         buildFullText(data.transcript ?? []),
    processing_status: 'pending',
    created_at:        new Date(),
  })

  // 3. Save MongoDB reference on meeting
  await prisma.meeting.update({
    where: { id: meeting.id },
    data:  { mongoTranscriptId },
  })

  // 4. Emit real-time event
  const { getIO } = await import('../../realtime/socket.server')
  getIO()?.to(`team:${meeting.teamId}`).emit('meeting:processing', {
    meetingId: meeting.id,
  })

  // 5. Push to extract queue
  await transcribeQueue.add('process-transcript', {
    meetingId:        meeting.id,
    teamId:           meeting.teamId,
    mongoTranscriptId,
  } satisfies ExtractJobData, {
    priority: 2,
  })
}

async function handleBotFailed(botId: string, reason: string): Promise<void> {
  const meeting = await findMeetingByBotId(botId)
  if (!meeting) return
  await updateMeetingStatus(meeting.id, 'FAILED', { processingError: reason })

  // Notify team via Socket.io
  const { getIO } = await import('../../realtime/socket.server')
  getIO()?.to(`team:${meeting.teamId}`).emit('meeting:failed', {
    meetingId: meeting.id,
    reason,
  })

  // Queue notification to team
  await notifyQueue.add('meeting-failed', {
    type:      'MEETING_FAILED' as any,
    teamId:    meeting.teamId,
    meetingId: meeting.id,
  })
}

// ── HELPERS ──────────────────────────────────────────────────────────────────

async function findMeetingByBotId(botId: string) {
  return prisma.meeting.findFirst({
    where: { recallBotId: botId },
  })
}

function buildFullText(transcript: RecallTranscriptTurn[]): string {
  return transcript
    .map((t) => `${t.speaker_name ?? t.speaker_tag} [${formatTime(t.start_time)}]: ${t.text}`)
    .join('\n')
}

function formatTime(seconds: number): string {
  const m = Math.floor(seconds / 60)
  const s = Math.floor(seconds % 60)
  return `${String(m).padStart(2, '0')}:${String(s).padStart(2, '0')}`
}


transcribe.worker.ts — Store Transcript

File: services/api/src/queues/workers/transcribe.worker.ts

typescriptimport { Worker, Job } from 'bullmq'
import { logger } from '../../config/logger'
import { mongoService } from '../../services/mongo.service'
import { extractQueue } from '../queue.client'
import { prisma } from '../../db/client'
import { ExtractJobData } from '../jobs/extract.job'

export const transcribeWorker = new Worker<ExtractJobData>(
  'transcribe',
  async (job: Job<ExtractJobData>) => {
    const { meetingId, teamId, mongoTranscriptId } = job.data
    logger.info({ jobId: job.id, meetingId }, 'transcribe.worker: processing')

    // Verify transcript exists in MongoDB
    const transcript = await mongoService.findTranscript(mongoTranscriptId)
    if (!transcript) {
      throw new Error(`Transcript not found in MongoDB: ${mongoTranscriptId}`)
    }

    // Fetch participants from PostgreSQL for speaker mapping
    const participants = await prisma.meetingParticipant.findMany({
      where: { meetingId },
    })

    // Build speaker tag → user mapping
    const speakerMap: Record<string, { userId?: string; name: string; email?: string }> = {}
    for (const p of participants) {
      if (p.speakerTag) {
        speakerMap[p.speakerTag] = {
          userId: p.userId ?? undefined,
          name:   p.name,
          email:  p.email ?? undefined,
        }
      }
    }

    // Enrich transcript turns with resolved speaker info
    const enrichedTurns = transcript.raw_transcript.map((turn: any) => ({
      ...turn,
      speaker_user_id: speakerMap[turn.speaker_tag]?.userId ?? null,
      speaker_name:    speakerMap[turn.speaker_tag]?.name ?? turn.speaker_tag,
      speaker_email:   speakerMap[turn.speaker_tag]?.email ?? null,
    }))

    // Update MongoDB with enriched transcript
    await mongoService.updateTranscript(mongoTranscriptId, {
      raw_transcript:    enrichedTurns,
      processing_status: 'ready_for_extraction',
    })

    // Push to extract queue
    await extractQueue.add(
      'extract-commitments',
      { meetingId, teamId, mongoTranscriptId } satisfies ExtractJobData,
      { priority: 2 }
    )

    logger.info({ jobId: job.id, meetingId }, 'transcribe.worker: done, pushed to extract queue')
  },
  {
    connection: { host: process.env.REDIS_HOST, port: parseInt(process.env.REDIS_PORT ?? '6379') },
    concurrency: 5,
  }
)

transcribeWorker.on('failed', (job, err) => {
  logger.error({ jobId: job?.id, err }, 'transcribe.worker: job failed')
})

Stripe Webhook Handler (Subscription Events)

File: services/api/src/modules/webhooks/stripe.webhook.ts

typescriptimport Stripe from 'stripe'

const stripe = new Stripe(process.env.STRIPE_SECRET_KEY!, { apiVersion: '2024-06-20' })

export async function handleStripeWebhook(req: Request, res: Response): Promise<void> {
  const sig = req.headers['stripe-signature'] as string
  let event: Stripe.Event

  try {
    event = stripe.webhooks.constructEvent(
      (req as any).rawBody,
      sig,
      process.env.STRIPE_WEBHOOK_SECRET!
    )
  } catch {
    res.status(400).json({ received: false, error: 'Invalid Stripe signature' })
    return
  }

  // Idempotency
  const idempotencyKey = `webhook:processed:stripe:${event.id}`
  if (await redis.exists(idempotencyKey)) {
    res.status(200).json({ received: true })
    return
  }
  await redis.setex(idempotencyKey, 86400, '1')

  res.status(200).json({ received: true })

  // Handle events asynchronously
  const obj = event.data.object as any

  switch (event.type) {
    case 'checkout.session.completed':
      await handleCheckoutCompleted(obj)
      break

    case 'customer.subscription.updated':
    case 'customer.subscription.created':
      await handleSubscriptionUpdated(obj)
      break

    case 'customer.subscription.deleted':
      await handleSubscriptionCancelled(obj)
      break

    case 'invoice.payment_succeeded':
      await handlePaymentSucceeded(obj)
      break

    case 'invoice.payment_failed':
      await handlePaymentFailed(obj)
      break
  }
}

async function handleSubscriptionUpdated(sub: Stripe.Subscription): Promise<void> {
  // Map Stripe price ID to Vocaply plan
  const priceId = sub.items.data[0]?.price.id ?? ''
  const plan    = mapPriceIdToPlan(priceId)

  await prisma.team.update({
    where: { stripeCustomerId: sub.customer as string },
    data:  {
      plan,
      stripeSubId:        sub.id,
      billingCycleStart:  new Date(sub.current_period_start * 1000),
      billingCycleEnd:    new Date(sub.current_period_end   * 1000),
    },
  })

  // Reset meetings_used counter on new billing cycle
  await prisma.team.update({
    where: { stripeCustomerId: sub.customer as string },
    data:  { meetingsUsed: 0 },
  })

  // Invalidate plan cache
  const team = await prisma.team.findFirst({ where: { stripeCustomerId: sub.customer as string } })
  if (team) await redis.del(`cache:team:plan:${team.id}`)
}

async function handleSubscriptionCancelled(sub: Stripe.Subscription): Promise<void> {
  await prisma.team.update({
    where: { stripeSubId: sub.id },
    data:  { plan: 'FREE', stripeSubId: null },
  })
}


BullBoard Setup (Job Monitoring UI)

File: services/api/src/config/bull-board.ts

typescriptimport { createBullBoard } from '@bull-board/api'
import { BullMQAdapter }   from '@bull-board/api/bullMQAdapter'
import { ExpressAdapter }  from '@bull-board/express'
import { transcribeQueue, extractQueue, notifyQueue, integrateQueue, deadlineQueue } from '../queues/queue.client'

export function setupBullBoard(app: Express) {
  if (process.env.NODE_ENV === 'production') return  // Disable in production by default

  const serverAdapter = new ExpressAdapter()
  serverAdapter.setBasePath('/admin/queues')

  createBullBoard({
    queues: [
      new BullMQAdapter(transcribeQueue),
      new BullMQAdapter(extractQueue),
      new BullMQAdapter(notifyQueue),
      new BullMQAdapter(integrateQueue),
      new BullMQAdapter(deadlineQueue),
    ],
    serverAdapter,
  })

  // Simple password protection for dev
  app.use('/admin/queues', (req, res, next) => {
    const auth = req.headers.authorization
    if (auth !== `Bearer ${process.env.BULL_BOARD_SECRET ?? 'dev-secret'}`) {
      res.status(401).send('Unauthorized')
      return
    }
    next()
  }, serverAdapter.getRouter())
}
