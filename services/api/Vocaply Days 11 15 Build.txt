6. Validate Middleware (Zod)

typescriptimport { AnyZodObject } from 'zod'

export function validate(schema: AnyZodObject) {
  return asyncHandler(async (req, _res, next) => {
    const result = schema.safeParse(req.body)
    if (!result.success) {
      const fields: Record<string, string> = {}
      result.error.errors.forEach((e) => {
        fields[e.path.join('.')] = e.message
      })
      throw new ValidationError(fields)
    }
    req.body = result.data  // Sanitized + coerced data
    next()
  })
}

10. Pagination Utility

typescript// Encode cursor: base64url({ id, createdAt })
export function encodeCursor(id: string, createdAt: Date): string {
  return Buffer.from(JSON.stringify({ id, createdAt: createdAt.toISOString() }))
    .toString('base64url')
}

// Decode cursor: base64url → { id, createdAt }
export function decodeCursor(cursor: string): { id: string; createdAt: Date } {
  const decoded = JSON.parse(Buffer.from(cursor, 'base64url').toString('utf8'))
  return { id: decoded.id, createdAt: new Date(decoded.createdAt) }
}

export function buildPaginationMeta(
  items: unknown[],
  limit: number,
  getCursorFields: (item: unknown) => { id: string; createdAt: Date }
) {
  const hasMore = items.length > limit
  const data = items.slice(0, limit)
  const nextCursor = hasMore
    ? encodeCursor(...Object.values(getCursorFields(data[data.length - 1])) as [string, Date])
    : null
  return { data, meta: { hasMore, nextCursor, count: data.length } }
}

typescriptimport jwt from 'jsonwebtoken'
import crypto from 'crypto'

export function generateAccessToken(user: User): string {
  return jwt.sign(
    {
      sub:    user.id,
      teamId: user.teamId,
      role:   user.role,
      email:  user.email,
    },
    process.env.JWT_SECRET!,
    {
      expiresIn: '15m',
      algorithm: 'HS256',
      issuer:    'vocaply.com',
      audience:  'vocaply-api',
    }
  )
}

export function generateRefreshToken(): string {
  // 64-char hex string — NEVER stored plain, only SHA-256 hash
  return crypto.randomBytes(32).toString('hex')
}

export function hashToken(token: string): string {
  return crypto.createHash('sha256').update(token).digest('hex')
}

export const COOKIE_OPTIONS = {
  httpOnly:  true,
  secure:    process.env.NODE_ENV === 'production',
  sameSite:  'strict' as const,
  path:      '/auth/refresh',   // Only sent to this path
  maxAge:    30 * 24 * 60 * 60 * 1000,  // 30 days
}

3. Auth Service — Register

File: services/api/src/modules/auth/auth.service.ts

register(data: { name, email, password }) flow:

1. Validate input (Zod schema handles this before controller)
2. existingUser = await repo.findByEmail(email.toLowerCase())
   → If found: throw DuplicateError('Email already registered')
3. passwordHash = await bcrypt.hash(password, 12)
4. user = await repo.create({ id: cuid(), email: email.toLowerCase(), name, passwordHash })
5. verificationToken = crypto.randomBytes(32).toString('hex')
6. tokenHash = sha256(verificationToken)
7. await repo.createEmailVerificationToken({ userId: user.id, tokenHash, expiresAt: +24h })
8. await emailService.sendVerificationEmail({ to: email, name, verificationToken })
9. return { message: "Check your email to verify your account", email }
   → NO user object, NO token in response (account unverified)

login(data: { email, password }) flow (brute force safe):

1. user = await repo.findByEmail(email.toLowerCase())
2. IF !user:
     await bcrypt.compare(password, FAKE_HASH)  // timing attack prevention
     throw UnauthorizedError('INVALID_CREDENTIALS', 'Invalid email or password')
3. IF user.lockedUntil && user.lockedUntil > new Date():
     throw RateLimitError('Account locked. Try again in X minutes.')
4. IF !user.emailVerified:
     throw ForbiddenError('EMAIL_NOT_VERIFIED') with message 'Please verify your email'
5. IF !user.passwordHash:
     throw UnauthorizedError('USE_OAUTH', 'This account uses Google/GitHub login')
6. passwordMatch = await bcrypt.compare(password, user.passwordHash)
7. IF !passwordMatch:
     newAttempts = user.failedAttempts + 1
     lockedUntil = newAttempts >= 5 ? addMinutes(new Date(), 15) : null
     await repo.updateFailedAttempts(user.id, newAttempts, lockedUntil)
     throw UnauthorizedError('INVALID_CREDENTIALS', 'Invalid email or password')
8. await repo.resetFailedAttempts(user.id)
9. await repo.updateLastLogin(user.id)
10. accessToken = generateAccessToken(user)
11. refreshToken = generateRefreshToken()
12. await repo.createRefreshToken({
      userId: user.id,
      tokenHash: hashToken(refreshToken),
      expiresAt: addDays(new Date(), 30),
      ipAddress: req.ip,
      userAgent: req.headers['user-agent']
    })
13. res.cookie('vocaply_refresh', refreshToken, COOKIE_OPTIONS)  // HttpOnly cookie
14. return { accessToken, user: { id, name, email, role, teamId, team } }

logout() flow:

1. refreshToken = req.cookies.vocaply_refresh
2. IF refreshToken:
     tokenHash = hashToken(refreshToken)
     await repo.findRefreshToken(tokenHash) → if found, delete it
3. res.clearCookie('vocaply_refresh', { path: '/auth/refresh' })
4. return { message: "Logged out successfully" }

7. requireAuth Middleware (Auth Middleware)

File: services/api/src/middleware/auth.middleware.ts

typescriptexport async function requireAuth(req, res, next) {
  const header = req.headers.authorization
  if (!header?.startsWith('Bearer ')) {
    return next(new UnauthorizedError('AUTH_REQUIRED', 'Authentication required'))
  }

  const token = header.split(' ')[1]
  try {
    const payload = jwt.verify(token, process.env.JWT_SECRET!, {
      algorithms: ['HS256'],
      issuer:     'vocaply.com',
      audience:   'vocaply-api',
    }) as JwtPayload

    req.user = {
      id:     payload.sub,
      teamId: payload.teamId,
      role:   payload.role,
      email:  payload.email,
    }
    next()
  } catch (err) {
    if (err instanceof jwt.TokenExpiredError) {
      return next(new UnauthorizedError('TOKEN_EXPIRED', 'Access token expired'))
    }
    return next(new UnauthorizedError('TOKEN_INVALID', 'Invalid access token'))
  }
}

8. newRefreshToken = generateRefreshToken()
9. await repo.createRefreshToken({
     userId: user.id,

8. Role Middleware

File: services/api/src/middleware/role.middleware.ts

typescriptconst ROLE_LEVELS: Record<string, number> = {
  OWNER: 4, ADMIN: 3, MANAGER: 2, MEMBER: 1
}

export function requireRole(...roles: string[]) {
  return (req: Request, _res: Response, next: NextFunction) => {
    const userLevel = ROLE_LEVELS[req.user.role] ?? 0
    const required  = Math.min(...roles.map((r) => ROLE_LEVELS[r] ?? 99))
    if (userLevel < required) {
      return next(new ForbiddenError(`Requires role: ${roles.join(' or ')}`))
    }
    next()
  }
}

Response interceptor — handles 401 with silent refresh:

typescript// Queues concurrent requests while refresh is in progress
let isRefreshing = false
let refreshSubscribers: Array<(token: string) => void> = []

axiosInstance.interceptors.response.use(
  (res) => res,
  async (error: AxiosError) => {
    const original = error.config as any
    if (error.response?.status === 401 && !original._retry) {
      if (isRefreshing) {
        return new Promise((resolve) => {
          refreshSubscribers.push((token) => {
            original.headers.Authorization = `Bearer ${token}`
            resolve(axiosInstance(original))
          })
        })
      }
      original._retry = true
      isRefreshing = true
      try {
        const res = await fetch('/api/auth/refresh', {
          method: 'POST', credentials: 'include'
        })
        if (!res.ok) throw new Error()
        const { accessToken } = await res.json()
        useAuthStore.getState().setAccessToken(accessToken)
        refreshSubscribers.forEach((cb) => cb(accessToken))
        refreshSubscribers = []
        original.headers.Authorization = `Bearer ${accessToken}`
        return axiosInstance(original)
      } catch {
        useAuthStore.getState().clearAuth()
        window.location.href = '/login?reason=session_expired'
      } finally {
        isRefreshing = false
      }
    }
    return Promise.reject(error)
  }
)

4. AuthProvider (Silent Refresh on Mount)

File: apps/web/src/shared/providers/AuthProvider.tsx

typescript'use client'

export function AuthProvider({ children }: { children: React.ReactNode }) {
  const { setUser, setAccessToken, setLoading, clearAuth } = useAuthStore()

  useEffect(() => {
    async function initAuth() {
      try {
        const res = await fetch('/api/auth/refresh', {
          method: 'POST',
          credentials: 'include',
        })
        if (!res.ok) { clearAuth(); return }
        const { accessToken, user } = await res.json()
        setAccessToken(accessToken)
        setUser(user)
      } catch {
        clearAuth()
      } finally {
        setLoading(false)
      }
    }
    initAuth()
  }, [])

  // Proactive refresh: 2 minutes before token expires (every 13 minutes)
  useEffect(() => {
    const interval = setInterval(async () => {
      if (!useAuthStore.getState().isAuthenticated) return
      try {
        const res = await fetch('/api/auth/refresh', { method: 'POST', credentials: 'include' })
        if (!res.ok) throw new Error()
        const { accessToken } = await res.json()
        setAccessToken(accessToken)
      } catch {
        clearAuth()
      }
    }, 13 * 60 * 1000)
    return () => clearInterval(interval)
  }, [])

  return <>{children}</>
}

5. BFF Route Handler (Silent Refresh Proxy)

File: apps/web/src/app/api/auth/refresh/route.ts

typescriptexport async function POST(req: NextRequest) {
  const cookie = req.cookies.get('vocaply_refresh')?.value
  if (!cookie) return NextResponse.json({ error: 'No refresh token' }, { status: 401 })

  // Proxy to backend API
  const backendRes = await fetch(`${process.env.API_URL}/auth/refresh`, {
    method:  'POST',
    headers: { Cookie: `vocaply_refresh=${cookie}` },
  })

  if (!backendRes.ok) {
    return NextResponse.json({ error: 'Refresh failed' }, { status: 401 })
  }

  const data   = await backendRes.json()
  const newCookie = backendRes.headers.get('set-cookie')

  const response = NextResponse.json(data)
  if (newCookie) response.headers.set('Set-Cookie', newCookie)
  return response
}
